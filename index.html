Few words about the company ...



Moxtra was founded in 2012 by Subrah Iyar and Stanley Huang. Both are seasoned veterans of the collaboration industry with decades of experience at both and Subrah was the co-founder and CEO of WebEx Communications and led the company from its inception to its sale to Cisco Systems in 2007 for $3.2 billion. Stanley was a Senior Director of Engineering at both WebEx and Cisco.

// 1. Setup your identity
// 2. Request a access token 


  function get_token() {
    var in_options = {
        uniqueid: "app_user_id",
        timestamp: timestamp,
        signature: signature,
        get_accesstoken: function(result) {
            alert("access_token: " + result.access_token + " expires in: " + result.expires_in);
        },
        error: function(result) {
            alert("error code: " + result.error_code + " message: " + result.error_message);
        }
    };
    Moxtra.setup(in_options);
}
// 3. You're all done, start the meet! 


function startMeet() {
    var options = {
        iframe: true,
        extension: { "show_dialogs": { "meet_invite": true } },
        tagid4iframe: "container",
        iframewidth: "1000px",
        iframeheight: "750px",
        access_token: token.value,
        error: function(event) {
            alert("error code: " + event.error_code + " message: " + event.error_message);
        }      
    };

    Moxtra.meet(options);
}
